<script type="text/javascript">
    //run only trusted code here :)
    function wikipediaFeedPostRender() {
        $('#wikipedia-feed').find('a').each(function (idx, anchor) {
            anchor.target = '_new';
            anchor.href = anchor.href.replace(window.location.href, 'http://en.wikipedia.org/');
        })
    }
    </script>
    <feed id="wikipedia-feed" count="10"
          src="http://en.wikipedia.org/w/api.php?action=featuredfeed&feed=featured&feedformat=atom"
          template-url="templates/wikipedia-featured-article.html"
          post-render="wikipediaFeedPostRender();"/>
